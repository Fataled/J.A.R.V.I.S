#!/bin/bash
# setup_models.sh - Downloads openWakeWord models for Jarvis wake word detection

set -e

MODELS_DIR="$(dirname "$0")/models"
BASE_URL="https://github.com/dscripka/openWakeWord/releases/download/v0.5.1"

echo "Creating models directory..."
mkdir -p "$MODELS_DIR"

declare -A MODELS=(
    ["hey_jarvis_v0.1.tflite"]="$BASE_URL/hey_jarvis_v0.1.tflite"
    ["melspectrogram.tflite"]="$BASE_URL/melspectrogram.tflite"
    ["embedding_model.tflite"]="$BASE_URL/embedding_model.tflite"
    ["silero_vad.onnx"]="$BASE_URL/silero_vad.onnx"
)

for filename in "${!MODELS[@]}"; do
    url="${MODELS[$filename]}"
    dest="$MODELS_DIR/$filename"

    if [ -f "$dest" ]; then
        echo "  [skip] $filename already exists"
    else
        echo "  [download] $filename..."
        if command -v curl &> /dev/null; then
            curl -L -o "$dest" "$url"
        elif command -v wget &> /dev/null; then
            wget -q -O "$dest" "$url"
        else
            echo "Error: neither curl nor wget found. Install one and retry."
            exit 1
        fi
        echo "  [done] $filename"
    fi
done

echo ""
echo "All models ready in $MODELS_DIR"
